function generateSessionToken(restaurantId, tableId) {
  const secret = process.env.SESSION_SECRET; 
  const data = `${restaurantId}-${tableId}-${Date.now()}`;
  const token = crypto.createHmac('sha256', secret).update(data).digest('hex');
  return token;
}
